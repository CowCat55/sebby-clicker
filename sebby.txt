Shinies
	*luckysebby
		on click:log Woop
		movement:onRight moveLeft fade bounce:0.05
		frequency:60
		frequency variation:30
		icon:stuff/https://download1652.mediafire.com/ykgg2x7sbzgg/9n9qb2yczt8o1tq/sebby.png
		class:bigButton
		on click:
			$boost=1
			if (have clover) $boost=4
			if (chance(25%))
				//get at least 7, or between 1 and 3 minutes of our sebby production
				$amount=max(7,random(sebbies:ps*60*1,sebbies:ps*60*3))*$boost
				toast The lucky sebby grants you<//><b>[$amount] sebbies</b>.
				yield $amount sebbies
			else
				$amount=1*$boost
				toast The lucky sebby grants you<//><b>[$amount] golden carrot[s?$amount]</b>!
				yield $amount goldenCarrot
			end
		end
